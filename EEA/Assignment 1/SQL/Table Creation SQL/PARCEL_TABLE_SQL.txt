DROP TABLE PARCEL;

CREATE TABLE PARCEL(
"PARCEL_ID" INT not null primary key 
GENERATED ALWAYS AS IDENTITY
(START WITH 1, INCREMENT BY 1),
"RECIPIENT_NAME" VARCHAR(255),
"ADDRESS_LINE_1" VARCHAR(255),
"ADDRESS_LINE_2" VARCHAR(255),
"POSTCODE" VARCHAR(255),
"CITY" VARCHAR(255),
"DELIVERED" BOOLEAN
);
