DROP TABLE FLIGHT;

CREATE TABLE FLIGHT(
"FLIGHT_ID" INT not null primary key 
GENERATED ALWAYS AS IDENTITY
(START WITH 1, INCREMENT BY 1),
"PLANE_ID" INT,
"DESTINATION" VARCHAR(50),
"ORIGIN" VARCHAR(50),
"DEPARTURE_DATE" DATE,
"DEPARTURE_TIME TIME,
"ARRIVAL_DATE" DATE,
"ARRIVAL_TIME" TIME,
"ETA" DATE,
"MAX_RISK" INT
FOREIGN KEY (PLANE_ID)
REFERENCES PLANE(PLANE_ID) ON DELETE CASCADE
);
