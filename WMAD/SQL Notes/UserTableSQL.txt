DROP TABLE USERS;

CREATE TABLE USERS(
"USER_ID" INT not null primary key 
GENERATED ALWAYS AS IDENTITY
(START WITH 1, INCREMENT BY 1),
"FORENAME" VARCHAR(20),
"SURNAME" VARCHAR(20),
"DOB" DATE,
"USERNAME" VARCHAR(50),
"PASSWORD" VARCHAR(64),
"ADDRESS_LINE_1" VARCHAR(75),
"ADDRESS_LINE_2" VARCHAR(75),
"POSTCODE" VARCHAR(10),
"ADMIN" BOOLEAN
);
